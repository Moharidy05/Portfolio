Project Overview

A comprehensive desktop application designed for the analysis, manipulation, and synthesis of digital signals. Unlike standard tools, this framework implements core DSP algorithms (like FFT, Convolution, and Quantization) from scratch to demonstrate a deep understanding of the underlying mathematics.

# Recursive FFT Implementation (From Scratch)
def _scratch_fft_recursive(x: np.ndarray) -> np.ndarray:
    """
    Computes the FAST Fourier Transform (FFT) 
    using the recursive Cooley-Tukey DIT algorithm.
    """
    x = np.asarray(x, dtype=complex)
    N = len(x)

    # Base Case
    if N == 1:
        return x

    # Recursive Step (Divide)
    x_even = _scratch_fft_recursive(x[::2])
    x_odd = _scratch_fft_recursive(x[1::2])

    # Twiddle factors (W_N^k)
    k_array = np.arange(N // 2)
    twiddles = np.exp(-2j * np.pi * k_array / N)

    # Combine (Butterfly Operation)
    t_product = twiddles * x_odd
    
    X = np.zeros(N, dtype=complex)
    X[:N//2] = x_even + t_product
    X[N//2:] = x_even - t_product

    return X
